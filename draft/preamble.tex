% ---- Section styling ----
% The spec is 0-indexed and carries its own manual numbers ("0.", "1.1") in the
% heading text, so we SUPPRESS LaTeX's own numbering and style the whole heading.
\setcounter{secnumdepth}{-2}
\titleformat{\section}
  {\needspace{4\baselineskip}\Large\bfseries\color{envoirViolet}}
  {}{0em}{}[{\vspace{0.2ex}\color{ruleGray}\titlerule[0.8pt]}]
\titleformat{\subsection}
  {\large\bfseries\color{inkGray}}{}{0em}{}
\titleformat{\subsubsection}
  {\normalsize\bfseries\color{envoirBlue}}{}{0em}{}
\titlespacing*{\section}{0pt}{2.4ex plus 1ex}{1.2ex}
% Sections still appear in the ToC (by title) even with numbering suppressed.
\setcounter{tocdepth}{2}
